## Phase 1：构建管线 + 首版 HTML

**目标**：Markdown → 完整 HTML 页面的全链路打通。

### 1.1 posts.js — 内容加载器

```js
// src/build/posts.js
import { readFile, readdir } from 'node:fs/promises';
import { join, extname } from 'node:path';
import matter from 'gray-matter';

const CONTENT_DIR = 'content/posts';

export async function loadPosts() {
  const files = await readdir(CONTENT_DIR);
  const mdFiles = files.filter(f => extname(f) === '.md');

  const posts = await Promise.all(
    mdFiles.map(async (filename) => {
      const raw = await readFile(join(CONTENT_DIR, filename), 'utf-8');
      const { data, content } = matter(raw);

      return {
        slug: data.slug || filename.replace('.md', ''),
        title: data.title || 'Untitled',
        date: data.date ? new Date(data.date) : new Date(),
        description: data.description || '',
        tags: data.tags || [],
        draft: data.draft || false,
        featured: data.featured || false,
        content,   // 原始 Markdown，稍后交给 render.js
        filename,
      };
    })
  );

  // 按日期倒序
  return posts
    .filter(p => !p.draft)
    .sort((a, b) => b.date - a.date);
}
```

### 1.2 render.js — Markdown → HTML

```js
// src/build/render.js
import MarkdownIt from 'markdown-it';
import { createHighlighter } from 'shiki';

let highlighter;

async function getHighlighter() {
  if (!highlighter) {
    highlighter = await createHighlighter({
      themes: ['github-light', 'github-dark'],
      langs: ['js', 'ts', 'css', 'html', 'bash', 'json', 'python', 'rust', 'go'],
    });
  }
  return highlighter;
}

const md = new MarkdownIt({
  html: false,        // 安全：不允许原始 HTML
  breaks: true,       // 换行转 <br>
  typographer: true,  // 智能引号、破折号
});

// 代码块高亮 —— 构建期完成
md.options.highlight = async (code, lang) => {
  const h = await getHighlighter();
  const html = h.codeToHtml(code, {
    lang: lang || 'text',
    themes: { light: 'github-light', dark: 'github-dark' },
  });
  return html;
};

export async function renderMarkdown(content) {
  return md.render(content);
}
```

**注意**：shiki 的 `codeToHtml` 是同步的，但如果用了 dual-theme，需要在构建脚本中用 `await` 模式。建议封装为异步函数，构建时逐篇调用。

### 1.4 index.js — 构建主入口

```js
// src/build/index.js
import { mkdir, writeFile, rm } from 'node:fs/promises';
import { loadPosts } from './posts.js';
import { renderMarkdown } from './render.js';
import { homePage, postPage, archivePage, layout } from './templates.js';

async function build() {
  console.time('build');

  // 1. 清空 dist
  await rm('dist', { recursive: true, force: true });
  await mkdir('dist/posts', { recursive: true });

  // 2. 加载所有文章
  const posts = await loadPosts();
  console.log(`Found ${posts.length} posts`);

  // 3. 渲染每篇文章
  for (const post of posts) {
    const body = await renderMarkdown(post.content);
    const html = postPage({ ...post, body });
    const full = layout({ title: post.title, description: post.description, content: html, page: 'post' });
    await writeFile(`dist/posts/${post.slug}.html`, full);
  }

  // 4. 生成首页
  const home = homePage(posts);
  await writeFile('dist/index.html', layout({ title: 'Blog', description: '', content: home, page: 'home' }));

  console.timeEnd('build');
}

build().catch(console.error);
```

### 3.1 main.js 功能清单（完整代码结构）

```js
// src/js/main.js（约 150 行，未压缩 ~15KB，gzip ~5KB）

// ===== 1. 主题切换 =====
function initTheme() {
  const stored = localStorage.getItem('theme');
  const prefers = window.matchMedia('(prefers-color-scheme: dark)').matches;
  const theme = stored || (prefers ? 'dark' : 'light');
  document.documentElement.setAttribute('data-theme', theme);
}

function toggleTheme() {
  const current = document.documentElement.getAttribute('data-theme');
  const next = current === 'dark' ? 'light' : 'dark';
  document.documentElement.setAttribute('data-theme', next);
  localStorage.setItem('theme', next);
}

// ===== 2. 阅读进度条 =====
function initProgressBar() {
  const bar = document.querySelector('[data-progress]');
  if (!bar) return;

  let ticking = false;
  window.addEventListener('scroll', () => {
    if (!ticking) {
      requestAnimationFrame(() => {
        const doc = document.documentElement;
        const max = doc.scrollHeight - doc.clientHeight;
        const pct = max > 0 ? doc.scrollTop / max : 0;
        bar.style.transform = `scaleX(${pct})`;
        ticking = false;
      });
      ticking = true;
    }
  }, { passive: true });
}

// ===== 3. 代码块复制 =====
function initCodeCopy() {
  document.querySelectorAll('pre').forEach((block) => {
    const btn = document.createElement('button');
    btn.className = 'code-copy';
    btn.textContent = 'Copy';
    btn.addEventListener('click', async () => {
      const code = block.querySelector('code')?.textContent || '';
      await navigator.clipboard.writeText(code);
      btn.textContent = 'Copied';
      setTimeout(() => { btn.textContent = 'Copy'; }, 1500);
    });
    block.appendChild(btn);
  });
}

// ===== 4. TOC 高亮（仅文章页） =====
function initTocHighlight() {
  const links = document.querySelectorAll('[data-toc] a');
  if (!links.length) return;

  const headings = Array.from(document.querySelectorAll('.prose h2, .prose h3'))
    .map(h => ({ id: h.id, el: h }));

  const observer = new IntersectionObserver((entries) => {
    for (const entry of entries) {
      if (entry.isIntersecting) {
        links.forEach(l => l.classList.remove('active'));
        const active = document.querySelector(`[data-toc] a[href="#${entry.target.id}"]`);
        active?.classList.add('active');
      }
    }
  }, { rootMargin: '-20% 0px -70% 0px' });

  headings.forEach(h => observer.observe(h.el));
}

// ===== 5. 链接预加载 =====
function initPrefetch() {
  const observer = new IntersectionObserver((entries) => {
    for (const entry of entries) {
      if (entry.isIntersecting) {
        const link = entry.target.querySelector('a');
        if (link) {
          const prefetch = document.createElement('link');
          prefetch.rel = 'prefetch';
          prefetch.href = link.href;
          document.head.appendChild(prefetch);
        }
        observer.unobserve(entry.target);
      }
    }
  });

  document.querySelectorAll('[data-prefetch]').forEach(el => observer.observe(el));
}

// ===== 启动 =====
document.addEventListener('DOMContentLoaded', () => {
  initTheme();
  initProgressBar();
  initCodeCopy();
  initTocHighlight();
  initPrefetch();

  document.querySelector('[data-theme-toggle]')?.addEventListener('click', toggleTheme);
});
```

### 4.1 图片管线（sharp）

```js
// src/build/images.js
import sharp from 'sharp';
import { mkdir } from 'node:fs/promises';
import { join } from 'node:path';

const SOURCE_DIR = 'images';
const OUTPUT_DIR = 'dist/assets/images';
const SIZES = [480, 800, 1200, 1600];
const FORMATS = ['webp', 'avif'];

export async function processImages() {
  await mkdir(OUTPUT_DIR, { recursive: true });

  // 遍历 SOURCE_DIR 所有图片
  for (const file of await readdir(SOURCE_DIR)) {
    if (!/\.(png|jpe?g|webp)$/i.test(file)) continue;
    const base = file.replace(/\.[^.]+$/, '');
    const input = join(SOURCE_DIR, file);

    for (const size of SIZES) {
      for (const fmt of FORMATS) {
        await sharp(input)
          .resize(size, undefined, { withoutEnlargement: true })
          [fmt]({ quality: fmt === 'avif' ? 50 : 75 })
          .toFile(join(OUTPUT_DIR, `${base}-${size}.${fmt}`));
      }
    }

    // 生成 20px blur placeholder（base64 内联）
    const blurBuf = await sharp(input).resize(20).webp({ quality: 20 }).toBuffer();
    const blurBase64 = `data:image/webp;base64,${blurBuf.toString('base64')}`;
    // 存到某个元数据文件供模板使用
  }
}
```

### 4.4 RSS / Sitemap

```js
// src/build/feed.js
export function generateRss(posts, site) {
  const items = posts.slice(0, 20).map(p => `
    <item>
      <title><![CDATA[${p.title}]]></title>
      <link>${site.url}/posts/${p.slug}</link>
      <description><![CDATA[${p.description}]]></description>
      <pubDate>${p.date.toUTCString()}</pubDate>
      <guid>${site.url}/posts/${p.slug}</guid>
    </item>`).join('');

  return `<?xml version="1.0" encoding="UTF-8"?>
<rss version="2.0">
  <channel>
    <title>${site.title}</title>
    <link>${site.url}</link>
    <description>${site.description}</description>
    <language>zh-CN</language>
    ${items}
  </channel>
</rss>`;
}

// src/build/sitemap.js
export function generateSitemap(posts, site) {
  const pages = [
    { loc: site.url, priority: '1.0', changefreq: 'daily' },
    ...posts.map(p => ({ loc: `${site.url}/posts/${p.slug}`, priority: '0.8' })),
  ];

  return `<?xml version="1.0" encoding="UTF-8"?>
<urlset xmlns="http://www.sitemaps.org/schemas/sitemap/0.9">
  ${pages.map(p => `<url><loc>${p.loc}</loc><priority>${p.priority}</priority></url>`).join('')}
</urlset>`;
}
```
